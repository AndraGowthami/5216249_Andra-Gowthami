#include<stdio.h>
#include<string.h>
#include<stdbool.h>
bool isPalindrome(char*s,int l,int r){
    while(l<r){
        if(s[l]!= s[r])
        return false;
        l++;
        r--;
    }
    return true;
}
int palindromeIndex(char*s){
    int l=0,r=strlen(s) - 1;
    while(l<r){
        if(s[l]!= s[r]){
            if(isPalindrome(s,l+1,r))
            return l;
            else if(isPalindrome(s,l,r-1))
            return r;
            else
            return -1;
        }
        l++;
        r--;
    }
